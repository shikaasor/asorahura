    ```typescript
    'use client';
    import { sendGAEvent } from '@next/third-parties/google';

    /**
     * Track a custom GA4 event.
     * In development: logs to console only (no data sent).
     * In production: sends to GA4 via sendGAEvent.
     *
     * Naming convention: {category}_{action}
     * e.g., assessment_click, assessment_start, checkout_begin
     */
    export function trackEvent(
      category: string,
      action: string,
      label?: string,
      value?: number
    ) {
      if (process.env.NODE_ENV !== 'production') {
        console.log(`[GA4 Event] ${category}_${action}`, { label, value });
        return;
      }

      const eventParams: Record<string, unknown> = {
        event_category: category,
        event_label: label,
      };
      if (value !== undefined) {
        eventParams.value = value;
      }

      sendGAEvent('event', `${category}_${action}`, eventParams);
    }
    ```

    Note: @next/third-parties is already included in Next.js 15 — no separate install needed for GA4.

    Create src/components/Analytics/ClarityProvider.tsx:

    ```typescript
    'use client';

    import { useEffect } from 'react';
    import { usePathname } from 'next/navigation';

    /**
     * Initializes Microsoft Clarity for heatmaps and session recording.
     * SCOPE: Only initializes on the 4 critical pages:
     *   - / (homepage)
     *   - /assessment
     *   - /results
     *   - /checkout
     * On all other routes this component returns null immediately with no side effects.
     * Per user decision: Clarity must NOT be initialized site-wide.
     *
     * Must be a client component — Clarity requires the window object.
     * Only runs in production when NEXT_PUBLIC_MICROSOFT_CLARITY_ID is set.
     */

    const CLARITY_PATHS = ['/', '/assessment', '/results', '/checkout'];

    export function ClarityProvider() {
      const pathname = usePathname();
      const clarityId = process.env.NEXT_PUBLIC_MICROSOFT_CLARITY_ID;

      useEffect(() => {
        // Hard gate: only initialize on the 4 critical pages
        if (!CLARITY_PATHS.includes(pathname)) return;
        if (!clarityId || process.env.NODE_ENV !== 'production') return;
        if (typeof window === 'undefined') return;

        // Avoid double initialization across route changes and hot reloads
        if ((window as Window & { clarity?: unknown }).clarity) return;

        (function (c: Window & { clarity?: unknown }, l: Document, a: string, r: string, i: string) {
          c[a as keyof typeof c] = c[a as keyof typeof c] || function (...args: unknown[]) {
            ((c[a as keyof typeof c] as { q?: unknown[] }).q = (c[a as keyof typeof c] as { q?: unknown[] }).q || []).push(args);
          };
          const t = l.createElement(r) as HTMLScriptElement;
          t.async = true;
          t.src = 'https://www.clarity.ms/tag/' + i;
          const y = l.getElementsByTagName(r)[0];
          y.parentNode?.insertBefore(t, y);
        })(window, document, 'clarity', 'script', clarityId);
      }, [pathname, clarityId]);

      return null;
    }
    ```

    Key changes from a naive global implementation:
    - `usePathname()` is called at component level (required — hooks cannot be inside useEffect).
    - `CLARITY_PATHS.includes(pathname)` is the first check inside useEffect — if the current page is not in the list, the function returns immediately and Clarity is never initialized.
    - `[pathname, clarityId]` dependency array ensures the gate re-evaluates on route changes.
    - The `window.clarity` guard still prevents double-injection if the user navigates away and back to a Clarity page.
