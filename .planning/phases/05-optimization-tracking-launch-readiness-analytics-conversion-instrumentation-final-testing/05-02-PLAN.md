    Create src/components/home/ScrollDepthTracker.tsx:

    ```typescript
    'use client';

    import { useEffect } from 'react';
    import { trackEvent } from '@/components/Analytics/trackEvent';

    const THRESHOLDS = [0.25, 0.5, 0.75, 1.0];

    /**
     * Fires engagement_scroll_depth events at 25%, 50%, 75%, 100% scroll milestones.
     * Tracked thresholds stored in module-level Set so they persist across re-renders
     * but reset on full page load (user refreshes = fresh session).
     * Renders nothing — side-effect only.
     */
    const tracked = new Set<number>();

    export function ScrollDepthTracker() {
      useEffect(() => {
        function handleScroll() {
          const scrolled = window.scrollY;
          const total = document.documentElement.scrollHeight - window.innerHeight;
          if (total <= 0) return;
          const ratio = scrolled / total;

          THRESHOLDS.forEach((threshold) => {
            if (ratio >= threshold && !tracked.has(threshold)) {
              tracked.add(threshold);
              trackEvent(
                'engagement',
                'scroll_depth',
                `${Math.round(threshold * 100)}pct`,
                Math.round(threshold * 100)
              );
            }
          });
        }

        window.addEventListener('scroll', handleScroll, { passive: true });
        return () => window.removeEventListener('scroll', handleScroll);
      }, []);

      return null;
    }
    ```
